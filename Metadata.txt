SELECT DISTINCT 
                         TOP (100) PERCENT SUBSTRING(dbo.ORDTASK.FOLDERNO, 0, 11) AS Muestra, dbo.CENTRAL.DATE_RECEIVED AS [Fecha Recibida], dbo.CENTRAL.STATUS AS Estado, 
                         dbo.CENTRAL.LAST_NAME AS [1er Apellido (bebé)], dbo.CENTRAL.MIDDLE_INITIAL AS [2do Apellido (Bebé)], dbo.CENTRAL.FIRST_NAME AS [Nombre (bebé)], 
                         dbo.METADATA_R_E.FIELD_VARCHAR_27 AS [1er Apellido Madre], dbo.METADATA_R_E.FIELD_VARCHAR_28 AS [2do Apellido Madre], 
                         dbo.METADATA_R_E.FIELD_VARCHAR_29 AS [Tipo de Id Madre], dbo.METADATA_R_E.FIELD_DATE_01 AS [Fecha Nacimiento Madre], 
                         dbo.METADATA_R_E.FIELD_VARCHAR_26 AS [Nombre Madre], dbo.CENTRAL.BIRTH_DATE AS [Fecha de Nacimiento], dbo.CENTRAL.DATE_COLLECTED AS [Fecha de Recolección], 
                         CONVERT(int, dbo.CENTRAL.DATE_COLLECTED - dbo.CENTRAL.BIRTH_DATE) AS [Edad de Toma], CONVERT(int, 
                         dbo.CENTRAL.DATE_RECEIVED - dbo.CENTRAL.DATE_COLLECTED) AS [Tiempo de Transito], dbo.CENTRAL.PREGNANCYWEEKS AS [Semanas de Embarazo], 
                         dbo.CENTRAL.WEIGHT AS [Peso al nacer], dbo.CENTRAL.TRANSFUSED AS [Px Tranfundido], dbo.CENTRAL.HOSPITALIZED AS [Px Hospitalizado], 
                         dbo.CENTRAL.TWIN_PATIENT AS [Px gemelar], dbo.CENTRAL.PATIENT_INTREATMENT AS [Px en tratamiento], dbo.CENTRAL.SEQ_OF_SAMPLE AS [Secuencia de muestra], 
                         dbo.METADATA_R_E.FIELD_VARCHAR_23 AS Provincia, dbo.METADATA_R_E.FIELD_VARCHAR_24 AS Canton, dbo.METADATA_R_E.FIELD_VARCHAR_25 AS Distrito, 
                         dbo.CENTRAL.REPETITION AS Repeticion, dbo.PACKAGES.CARRIER AS [VIA DE INGRESO], dbo.CENTRAL.ORIGREC AS REGISTRO, dbo.CENTRAL.USER_DE AS USUARIO1DATA, 
                         dbo.CENTRAL.USER_DBDE AS USUARIO2DATA, dbo.CENTRAL.USER_DBDECONF AS USUARIO_CONFIRMA, dbo.RASCLIENTS.CITY AS REGION, 
                         dbo.RASCLIENTS.CATEGORY AS CATEGORIA, dbo.RASCLIENTS.COMPNAME AS [Centro de Recolección], dbo.SAMPLE_SUMMARY.WARNCODE, dbo.SAMPLE_SUMMARY.CLASSIFICATION, 
                         dbo.SAMPLE_SUMMARY.SUB_CLASSIFICATION, dbo.SAMPLE_SUMMARY.TESTGROUPNAME, dbo.METADATA_R_E.FIELD_VARCHAR_35 AS [Tomada por], dbo.RASCLIENTS.RASCLIENTID AS IDCentro, 
                         dbo.RASCLIENTS.HEALTH_AREA_CODE AS IDArea, dbo.ORDTASK.FOLDERNO AS [Numero de Folder], dbo.CENTRAL.RECEIPT_NUMBER AS BoletaNo, dbo.CENTRAL.PATIENT_ID AS id, 
                         dbo.CENTRAL.SEX, dbo.METADATA_R_E.FIELD_VARCHAR_41 AS [Leche Materna], dbo.METADATA_R_E.FIELD_VARCHAR_42 AS Formula, 
                         dbo.METADATA_R_E.FIELD_VARCHAR_43 AS Mixta, dbo.METADATA_R_E.FIELD_VARCHAR_44 AS [Leche de Vaca], dbo.METADATA_R_E.FIELD_VARCHAR_45 AS Parenteral, 
                         dbo.METADATA_R_E.FIELD_VARCHAR_46 AS Otra, dbo.METADATA_R_E.FIELD_VARCHAR_20 AS Hosp_Nacimiento, dbo.METADATA_R_E.FIELD_VARCHAR_33 AS OBSERVACIONES, 
                         dbo.METADATA_R_E.FIELD_VARCHAR_30 AS IDMADRE
FROM            dbo.ORDTASK INNER JOIN
                         dbo.CENTRAL ON SUBSTRING(dbo.ORDTASK.FOLDERNO, 0, 11) = dbo.CENTRAL.EXTERNAL_ID INNER JOIN
                         dbo.METADATA_R_E ON dbo.METADATA_R_E.ORIGINAL_ORIGREC = dbo.CENTRAL.ORIGREC INNER JOIN
                         dbo.RASCLIENTS ON dbo.CENTRAL.RASCLIENTID = dbo.RASCLIENTS.RASCLIENTID INNER JOIN
                         dbo.PACKAGES ON dbo.CENTRAL.PACKAGEID = dbo.PACKAGES.PACKAGEID INNER JOIN
                         dbo.SAMPLE_SUMMARY ON dbo.CENTRAL.EXTERNAL_ID = dbo.SAMPLE_SUMMARY.EXTERNAL_ID
WHERE        (dbo.CENTRAL.DATE_RECEIVED >= CONVERT(DATETIME, '2019-01-01 00:00:00', 120)) AND (dbo.CENTRAL.DATE_RECEIVED <= CONVERT(DATETIME, '2019-12-31 23:59:59', 120)) AND 
                         (dbo.METADATA_R_E.FIELD_VARCHAR_26 <> 'NULL') AND (dbo.CENTRAL.SP_CODE = 676)
GROUP BY dbo.CENTRAL.ORIGREC, dbo.ORDTASK.FOLDERNO, dbo.CENTRAL.DATE_RECEIVED, dbo.CENTRAL.BIRTH_DATE, dbo.CENTRAL.DATE_COLLECTED, 
                         dbo.CENTRAL.LAST_NAME, dbo.CENTRAL.MIDDLE_INITIAL, dbo.CENTRAL.FIRST_NAME, dbo.METADATA_R_E.FIELD_VARCHAR_27, 
                         dbo.METADATA_R_E.FIELD_VARCHAR_28, dbo.METADATA_R_E.FIELD_VARCHAR_26, dbo.ORDTASK.TESTNO, dbo.CENTRAL.STATUS, 
                         dbo.CENTRAL.PREGNANCYWEEKS, dbo.CENTRAL.WEIGHT, dbo.CENTRAL.TRANSFUSED, dbo.CENTRAL.HOSPITALIZED, dbo.CENTRAL.TWIN_PATIENT, 
                         dbo.CENTRAL.PATIENT_INTREATMENT, dbo.CENTRAL.SEQ_OF_SAMPLE, dbo.RASCLIENTS.COMPNAME, dbo.METADATA_R_E.FIELD_VARCHAR_28, 
                         dbo.METADATA_R_E.FIELD_VARCHAR_23, dbo.METADATA_R_E.FIELD_VARCHAR_24, dbo.METADATA_R_E.FIELD_VARCHAR_25, dbo.METADATA_R_E.FIELD_DATE_01, 
                         dbo.METADATA_R_E.FIELD_VARCHAR_29, dbo.CENTRAL.REPETITION, dbo.PACKAGES.CARRIER, dbo.CENTRAL.USER_DE, dbo.CENTRAL.USER_DBDE, 
                         dbo.CENTRAL.USER_DBDECONF, dbo.RASCLIENTS.CITY, dbo.RASCLIENTS.CATEGORY, dbo.SAMPLE_SUMMARY.WARNCODE, dbo.SAMPLE_SUMMARY.CLASSIFICATION, 
                         dbo.SAMPLE_SUMMARY.SUB_CLASSIFICATION, dbo.SAMPLE_SUMMARY.TESTGROUPNAME, dbo.METADATA_R_E.FIELD_VARCHAR_35, dbo.RASCLIENTS.RASCLIENTID, 
                         dbo.RASCLIENTS.HEALTH_AREA_CODE, dbo.CENTRAL.RECEIPT_NUMBER, dbo.CENTRAL.PATIENT_ID, dbo.CENTRAL.SEX, dbo.METADATA_R_E.FIELD_VARCHAR_41, 
                         dbo.METADATA_R_E.FIELD_VARCHAR_42, dbo.METADATA_R_E.FIELD_VARCHAR_43, dbo.METADATA_R_E.FIELD_VARCHAR_44, dbo.METADATA_R_E.FIELD_VARCHAR_45, 
                         dbo.METADATA_R_E.FIELD_VARCHAR_46, dbo.METADATA_R_E.FIELD_VARCHAR_20, dbo.METADATA_R_E.FIELD_VARCHAR_33, 
                         dbo.METADATA_R_E.FIELD_VARCHAR_30
ORDER BY REGISTRO
